(defun TcfBeardBuilderCB (f)
  (let (cvId fillerCv xfiller yfiller xf yf x y gapCv lowerXAlign lowerYAlign)

  (setq xf (atof f->xf->value))
 (setq yf (atof f->yf->value))

  (setq fillerCv (dbOpenCellViewByType (TcfXmlFind (TcfGetProjectName) "stacked_filler" "Library")
  (TcfXmlFind (TcfGetProjectName) "stacked_filler" "CellName")
  "layout" "maskLayout" "r"))

 (setq yfiller (TcfGetYdim fillerCv
   ?orientation (TcfXmlFind (TcfGetProjectName) "stacked_filler" "Orientation")))
 (setq xfiller (TcfGetXdim fillerCv
   ?orientation (TcfXmlFind (TcfGetProjectName) "stacked_filler" "Orientation")))

  (if (equal f->xfunits->value "microns") (setq x (sub1 (round (quotient xf xfiller)))) (setq x (sub1 (round xf))))
 (if (equal f->yfunits->value "microns") (setq y (sub1 (round (quotient yf yfiller)))) (setq y (sub1 (round yf))))

  (when (equal f->measuredRegion->value "Outer region")
   (setq x (difference x 2))
   (setq y (difference y 2))
  ) 
  (setq cvId (dbOpenCellViewByType f->libf->value f->cellf->value "layout" "maskLayout" "w"))
 (setq lowerXAlign (minus (times (quotient xfiller 2) (plus x 2))))
 (setq lowerYAlign (minus (times (quotient yfiller 2) (plus y 2))))
 (TcfCreateContourFrame (symbolToString (concat "E" x " N" y " W" x " S" y))
   "beard"
   f->innerkeep->value
   f->outerkeep->value
   (TcfGetProjectName)
   cvId
   ?lowerXAlign lowerXAlign
   ?lowerYAlign lowerYAlign)

  (if f->addpycheck->value
  (TcfCreateInstance cvId
   (TcfXmlFind (TcfGetProjectName) "filler_polycheck" "Library")
   (TcfXmlFind (TcfGetProjectName) "filler_polycheck" "CellName")
   ?Orientation (TcfXmlFind (TcfGetProjectName) "filler_polycheck" "Orientation")
   ?MosaicX (plus x 1)
   ?MosaicY (plus y 1)
   ?lowerXAlign (minus (times (quotient xfiller 2) (plus x 1)))
   ?lowerYAlign (minus (times (quotient yfiller 2) (plus y 1))))
  ) 
 (dbClose fillerCv)
 (dbClose cvId)

 t
  ))

 (TccInfo "TcfEOWFrameBuilderCore" (sprintf nil "Starting core routine for frame type=%L." frametype))
  (setq fillerCv (dbOpenCellViewByType (TcfXmlFind (TcfGetProjectName) "stacked_filler" "Library")
  (TcfXmlFind (TcfGetProjectName) "stacked_filler" "CellName")

 "layout" "maskLayout" "r"))

 (setq leftcornerCv (dbOpenCellViewByType (TcfXmlFind (TcfGetProjectName) (strcat "eow_" (if (equal frametype "Container") "left" (TcfFrameCellLocationReflect "left"))) "Library")
   (TcfXmlFind (TcfGetProjectName) (strcat "eow_" (if (equal frametype "Container") "left" (TcfFrameCellLocationReflect "left"))) "CellName")
   "layout" "maskLayout" "r"))

  "layout" "maskLayout" "r"))

   ?orientation (TcfXmlFind (TcfGetProjectName) "stacked_filler" "Orientation")))
 (setq xfiller (TcfGetXdim fillerCv
   ?orientation (TcfXmlFind (TcfGetProjectName) "stacked_filler" "Orientation")))

 (setq leftheight (TcfGetYdim leftcornerCv
   ?orientation (TcfXmlFind (TcfGetProjectName) (strcat "eow_" (if (equal frametype "Container") "left" (TcfFrameCellLocationReflect "left"))) "Orientation")))
 (setq leftwidth (TcfGetXdim leftcornerCv
  ?orientation (TcfXmlFind (TcfGetProjectName) (strcat "eow_" (if (equal frametype "Container") "left" (TcfFrameCellLocationReflect "left"))) "Orientation")))

 (setq bottomwidth (TcfGetXdim bottomcornerCv
 ?orientation (TcfXmlFind (TcfGetProjectName) (strcat "eow_" (if (equal frametype "Container") "bottom" (TcfFrameCellLocationReflect "bottom"))) "Orientation")))
 (setq bottomheight (TcfGetYdim bottomcornerCv
 ?orientation (TcfXmlFind (TcfGetProjectName) (strcat "eow_" (if (equal frametype "Container") "bottom" (TcfFrameCellLocationReflect "bottom"))) "Orientation")))

 leftheight))))
