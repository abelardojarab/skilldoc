(defun TcfXmlFind (@key (root nil) @rest tags)
  "This function performs multi-search on XML data."
  (let (data tag result libaliases)
 (importSkillVar TCC_GLOBAL_VARS)

  (if root (setq data root))

  (unless data
   (if (boundp 'TCC_GLOBAL_VARS)
 (setq data TCC_GLOBAL_VARS->TcfXmlData)
  (progn
 (importSkillVar TCC_CONTROLS_FILE)
 (setq data TCC_CONTROLS_FILE->TcfXmlData)
  )) 
   (unless data
  (setq data (TcfXmlLoad))
  (if (boundp 'TCC_GLOBAL_VARS) TCC_GLOBAL_VARS->TcfXmlData = data
   TCC_CONTROLS_FILE->TcfXmlData = data
  )) ) 
 (foreachs tag tags
  (when (and data tag)
 (setq data (TcfXmlFindTagBFS data tag))
  )) 
 (if (listp (nth 2 data))
  (setq result data)
   (setq result (tcfatot (nth 2 data) (nth 1 data)->type))
  ) 
  (importSkillVar TCC_GLOBAL_VARS)
 (if (boundp 'TCC_GLOBAL_VARS)
  (foreachs elem (TccFlattenList TCC_GLOBAL_VARS->TccLibs)
   (unless (member elem TCC_GLOBAL_VARS->TccLibs)
  (setq libaliases (cons elem libaliases))
  )) ) (if (member result libaliases) (setq result (TccLib result))) (unless result
   (TccInfo "TcfXmlFind" (sprintf nil "Returning nil for tags = %L" tags))
  ) 
 result
  )) 
 
(defun TcfXmlFindTagBFS (root tag)
  "Using BFS to look a tag in the XML tree"
  (let ( (prunedTree nil) elem queue xmltree)

  (setq queue (list root))

  (while (length queue)!=0
  (setq xmltree (car queue)) (setq queue (cdr queue)) 
  (when (listp xmltree)
 (if (equal (symbolToString (nth 0 xmltree)) tag)
  prunedTree = xmltree
  )) 
  (if prunedTree
   queue = nil 
   (when (listp xmltree)
  (when  (leqp 2 (length xmltree))
 (foreachs elem (cddr xmltree) queue = (append queue (list elem)) )) )) ) 
 prunedTree
  ))

(defun TcfGetFillerCellInfo (layerName @key (customCell nil) (customLib nil))
  "This function outputs the distance to span a given number tracks for a filler cell"
  (let (tmpfCv fillerCellName horizBBoxes vertBBoxes lstXPoints lstYPoints lstXPoints lstYPoints chkBoundarybBox xlower ylower xupper yupper isPGD (k 0) (j 0) (isPGD nil) l w lstPoints lstCDSpace shapes layersList visitedj comPat2v comPat2m shapetransform shapedb shapesBBoxes techId lstLayers lstLengths offsetETE offsetSides shapesBBoxesCopy1)

 (importSkillVar TCC_GLOBAL_VARS)
 (if (and (stringp (car (gethash layerName TCC_GLOBAL_VARS->TcfFillerCellsCache))) (not customCell))
  (gethash layerName TCC_GLOBAL_VARS->TcfFillerCellsCache) (progn
  (setq comPat2v (pcreCompile "^via")) (setq comPat2m (pcreCompile "^metal[cs]*")) 
 (if (and (stringp customLib) (stringp customCell))
   (if (ddGetObj customLib customCell) (progn
   (setq fillerCellName customCell)
   (TccInfo "TcfGetFillerCellInfo" (sprintf nil "Using custom filler cell = %L" customCell))
  ) (TccWarn "TcfGetFillerCellInfo" "Custom cell does not exist!")
  )) 
  (unless fillerCellName
 (setq fillerCellName (car (TcfGetLstFillerCells layerName)))
  ) 
  (if customLib
   (if (ddGetObj customLib) (setq tmpfCv (dbOpenCellViewByType customLib fillerCellName "layout"))

 (list (pcreReplace comPat2v layerName "viaa" 0) (TcfGetDefaultLPPPurpose ?cvId tmpfCv))
 (list (pcreReplace comPat2v layerName "viab" 0) (TcfGetDefaultLPPPurpose ?cvId tmpfCv))
 (list (pcreReplace comPat2v layerName "viac" 0) (TcfGetDefaultLPPPurpose ?cvId tmpfCv))))

  (setq techId (techGetTechFile tmpfCv))
  (setq lstLayers (TcfRemoveLstDuplicateItems (setofs x lstLayers (techGetLP techId x))))

  (TccInfo "TcfGetFillerCellInfo" (sprintf nil "Scanning filler cell with shapes on following layers: %L" lstLayers))
  (setq shapes (TcfGetTrueOverlaps tmpfCv (getSGq tmpfCv bBox) (car lstLayers) (dbGetMaxHierDepth) ?lstLayers lstLayers))
  (setq shapesBBoxes (mapcar (lambda (x) (geTransformUserBBox (car x)~>bBox (cadr x))) shapes))

 (setq shapesBBoxesCopy1 shapesBBoxes)
  (if shapes
   (progn

 (setq horizBBoxes shapesBBoxes)
  (setq vertBBoxes shapesBBoxes)

 (foreachs elemx1 horizBBoxes
   (setq lstXPoints (append lstXPoints (list (xCoord (lowerLeft elemx1)) (xCoord (upperRight elemx1)))))
  ) (foreachs elemx2 vertBBoxes
   (setq lstYPoints (append lstYPoints (list (yCoord (lowerLeft elemx2)) (yCoord (upperRight elemx2)))))
  ) 
  (setq lstXPoints (TcfRemoveLstDuplicateItems (sort lstXPoints 'greaterp)))
  (setq lstYPoints (TcfRemoveLstDuplicateItems (sort lstYPoints 'greaterp)))

 chkBoundarybBox = (TcfGetBoundingBox tmpfCv)
  xlower = (xCoord (lowerLeft chkBoundarybBox))
  ylower = (yCoord (lowerLeft chkBoundarybBox))
  xupper = (xCoord (upperRight chkBoundarybBox))
  yupper = (yCoord (upperRight chkBoundarybBox))

 (setq l (cadr (trBBoxDims (car horizBBoxes))))
  (setq w (car (trBBoxDims (car horizBBoxes))))
  (if (greaterp l w) (setq isPGD t) (setq isPGD nil))

 (setq horizBBoxes (sort shapesBBoxes 'TccSortBBoxRowX))
  (setq vertBBoxes (sort shapesBBoxesCopy1 'TccSortBBoxRowY))

  (setq shapetransform (cadr shape_elem))
  (when (and (TccEQP (mycoord (lowerLeft (geTransformUserBBox (getSGq shapedb bBox) shapetransform))) (nth j lstPoints)) (not (member j visitedj)))
 (setq layersList (cons (car (getSGq shapedb lpp)) layersList))

  )) (setq j (add1 j))
  ) (setq layersList (reverse layersList))

 (setq offsetETE
  (if isPGD
   (list (difference (yCoord (lowerLeft (car vertBBoxes))) ylower)
   (difference yupper (yCoord (upperRight (car (reverse vertBBoxes)))))

 (difference xupper (xCoord (upperRight (car (reverse horizBBoxes)))))
  ))
  ) 
  (setq offsetSides
  (if isPGD
   (list (difference (xCoord (lowerLeft (car horizBBoxes))) xlower)
   (difference xupper (xCoord (upperRight (car (reverse horizBBoxes)))))
  ) (list (difference (yCoord (lowerLeft (car vertBBoxes))) ylower)
 (difference yupper (yCoord (upperRight (car (reverse vertBBoxes)))))
  ))
  ) 
 (foreachs shape horizBBoxes
   (if isPGD
 (setq lstLengths (cons (TccPutOnGrid (cadr (trBBoxDims shape))) lstLengths))
  (setq lstLengths (cons (TccPutOnGrid (car (trBBoxDims shape))) lstLengths))
  )) (setq lstLengths (sort lstLengths 'greaterp))
  (setq lstLengths (TcfRemoveLstDuplicateItems lstLengths))

 (dbClose tmpfCv)

 (list layerName
  layersList
  lstCDSpace
  (list (difference xupper xlower)
  (difference yupper ylower)) (list (car offsetETE) (car offsetSides)) (if isPGD "pgd" "ogd") lstLengths   offsetETE   offsetSides   )) nil
  )) )) )
